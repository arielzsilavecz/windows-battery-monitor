# COMANDOS PARA SUBIR A GITHUB
# Copia y pega estos comandos uno por uno en PowerShell

# 1. Inicializar repositorio Git
git init

# 2. Agregar todos los archivos
git add .

# 3. Hacer el primer commit
git commit -m "Initial commit: Windows Battery Alert Monitor"

# 4. Conectar con tu repositorio de GitHub
# REEMPLAZA 'tu-usuario' y 'nombre-repo' con los tuyos
git remote add origin https://github.com/tu-usuario/nombre-repo.git

# 5. Subir archivos a GitHub
git push -u origin main

# Si da error con 'main', prueba con:
# git branch -M main
# git push -u origin main

echo "¡LISTO! Tu proyecto ya está en GitHub"
